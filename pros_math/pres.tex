\begin{frame}{Toeplitz Matrix}
\[
h[i-j] := a_{ij}
\]
\[
(x \star h)[i] = \sum\limits_{j=0}^{N-1} x[j] \cdot h[i-j] = b[i]
\]
\[
\begin{array}{c|cccc}
b[i]/x[j] & x[0] & x[1] & ... & x[N-1]  \\
\hline
b[0]     & h[0] & h[-1] & \cdots & h[-N+1] \\
b[1]     & h[1] & h[0] & \cdots & h[-N+2]  \\
\vdots  & \vdots & \vdots & \ddots & \vdots \\
b[N-1] & h[N-1] & h[N-2] & \cdots & h[0]  \\
\end{array}
\]
\end{frame}
